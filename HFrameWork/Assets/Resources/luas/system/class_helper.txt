





local classHelper = {}

function classHelper.declare(className, super)
local class = {}
class.__cname = className
if super then
setmetatable(class, super)
class.base = super
end
class.__index = class
function class:new(...)
local ins = {}
setmetatable(ins, class)
ins:Ctor(...)
return ins
end

return class
end

return classHelper
